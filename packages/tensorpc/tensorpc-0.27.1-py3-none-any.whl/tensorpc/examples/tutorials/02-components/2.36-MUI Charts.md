## MUI Chart

MUI Chart is a set of chart components based on [MUI X](https://mui.com/x/react-charts/).

### Bar Chart

```Python
from tensorpc.dock import mui, mark_create_layout, chart
class App:
    @mark_create_layout
    def my_layout(self):
        self.plot = chart.BarChart().prop(
            series=[
                chart.BarSeries(data=[4, 3, 5]),
                chart.BarSeries(data=[1, 6, 3]),
                chart.BarSeries(data=[2, 5, 6]),
            ],
            xAxis=[chart.XAxis(data=["A", "B", "C"])],
        )
        self.plot.event_item_click.on(lambda x: print(x))
        return mui.VBox([
            self.plot,
        ]).prop(flex=1)

```

### Scatter Chart

```Python
from tensorpc.dock import mui, mark_create_layout, chart
import numpy as np 
class App:
    @mark_create_layout
    def my_layout(self):
        xs = np.arange(0, 50) / 50 * 3.1416
        ys1 = np.sin(xs)
        ys2 = np.cos(xs)

        self.plot = chart.ScatterChart().prop(
            series=[
                chart.ScatterSeries(label="A", data=[{"x": xs[i], "y": ys1[i]} for i in range(len(xs))]),
                chart.ScatterSeries(label="B", data=[{"x": xs[i], "y": ys2[i]} for i in range(len(xs))]),
            ],
        )
        return mui.VBox([
            self.plot,
        ]).prop(flex=1)

```
